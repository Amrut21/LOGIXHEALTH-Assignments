1. Create a Table With Multiple Foreign Keys.

CREATE TABLE Customer(
CustomerId int primary key,
fname varchar(45),
lname varchar(45),
age int,
country varchar(45)
)

CREATE TABLE Orders(
orderId int primary key,
item varchar(25) ,
amount int,
customerId int,
foreign key (customerId) references Customer(CustomerId)
)

CREATE TABLE SHIPPING (
shippingId int primary key,
status VARCHAR(20),
customerId int references Customer(CustomerId),
)

CREATE TABLE summery(
summeryId int,
customerId int references Customer(CustomerId),
orderId int references Orders(orderID),
shippingId int references SHIPPING(shippingId)
)


2. Create a table namely Child (fname, lname, address, pin).
 This table has the foreign key ''pin'' which references the ''pin'' 
column in the Parent table. Make sure when you delete a record in the parent table, 
the records that contain the same value in the pin column of the Child table will be deleted also.

CREATE TABLE Parent(
pin int primary key,
Name varchar(45),
pno int,
address varchar(45)
)

CREATE TABLE Child(
fname varchar(45),
lname varchar(45),
address varchar(90),
pin int references Parent(pin)
on update cascade on delete cascade
)
